47 inline pid_t defaultClone(const lambda::function<int()>& func)
48 {
49  pid_t pid = ::fork();
50  if (pid == -1) {
51  return -1;
52  } else if (pid == 0) {
53  // Child.
54  ::exit(func());
55  UNREACHABLE();
56  } else {
57  // Parent.
58  return pid;
59  }
60 }

65 // This function will invoke `os::cloexec` on all specified file
66 // descriptors that are valid (i.e., not `None` and >= 0).
67 inline Try<Nothing> cloexec(
68  const InputFileDescriptors& stdinfds,
69  const OutputFileDescriptors& stdoutfds,
70  const OutputFileDescriptors& stderrfds)
71 {
72  hashset<int> fds = {
73  stdinfds.read,
74  stdinfds.write.getOrElse(-1),
75  stdoutfds.read.getOrElse(-1),
76  stdoutfds.write,
77  stderrfds.read.getOrElse(-1),
78  stderrfds.write
79  };
80 
81  foreach (int fd, fds) {
82  if (fd >= 0) {
83  Try<Nothing> cloexec = os::cloexec(fd);
84  if (cloexec.isError()) {
85  return Error(cloexec.error());
86  }
87  }
88  }
89 
90  return Nothing();
91 }

103 // The main entry of the child process.
104 //
105 // NOTE: This function has to be async signal safe.
106 inline int childMain(
107  const std::string& path,
108  char** argv,
109  char** envp,
110  const InputFileDescriptors& stdinfds,
111  const OutputFileDescriptors& stdoutfds,
112  const OutputFileDescriptors& stderrfds,
113  bool blocking,
114  int pipes[2],
115  const std::vector<Subprocess::ChildHook>& child_hooks)
116 {
117  // Close parent's end of the pipes.
118  if (stdinfds.write.isSome()) {
119  ::close(stdinfds.write.get());
120  }
121  if (stdoutfds.read.isSome()) {
122  ::close(stdoutfds.read.get());
123  }
124  if (stderrfds.read.isSome()) {
125  ::close(stderrfds.read.get());
126  }
127 
128  // Currently we will block the child's execution of the new process
129  // until all the parent hooks (if any) have executed.
130  if (blocking) {
131  ::close(pipes[1]);
132  }
133 
134  // Redirect I/O for stdin/stdout/stderr.
135  while (::dup2(stdinfds.read, STDIN_FILENO) == -1 && errno == EINTR);
136  while (::dup2(stdoutfds.write, STDOUT_FILENO) == -1 && errno == EINTR);
137  while (::dup2(stderrfds.write, STDERR_FILENO) == -1 && errno == EINTR);
138 
139  // Close the copies. We need to make sure that we do not close the
140  // file descriptor assigned to stdin/stdout/stderr in case the
141  // parent has closed stdin/stdout/stderr when calling this
142  // function (in that case, a dup'ed file descriptor may have the
143  // same file descriptor number as stdin/stdout/stderr).
144  //
145  // We also need to ensure that we don't "double close" any file
146  // descriptors in the case where one of stdinfds.read,
147  // stdoutfds.write, or stdoutfds.write are equal.
148  if (stdinfds.read != STDIN_FILENO &&
149  stdinfds.read != STDOUT_FILENO &&
150  stdinfds.read != STDERR_FILENO) {
151  ::close(stdinfds.read);
152  }
153  if (stdoutfds.write != STDIN_FILENO &&
154  stdoutfds.write != STDOUT_FILENO &&
155  stdoutfds.write != STDERR_FILENO &&
156  stdoutfds.write != stdinfds.read) {
157  ::close(stdoutfds.write);
158  }
159  if (stderrfds.write != STDIN_FILENO &&
160  stderrfds.write != STDOUT_FILENO &&
161  stderrfds.write != STDERR_FILENO &&
162  stderrfds.write != stdinfds.read &&
163  stderrfds.write != stdoutfds.write) {
164  ::close(stderrfds.write);
165  }
166 
167  if (blocking) {
168  // Do a blocking read on the pipe until the parent signals us to
169  // continue.
170  char dummy;
171  ssize_t length;
172  while ((length = ::read(pipes[0], &dummy, sizeof(dummy))) == -1 &&
173  errno == EINTR);
174 
175  if (length != sizeof(dummy)) {
176  ABORT("Failed to synchronize with parent");
177  }
178 
179  // Now close the pipe as we don't need it anymore.
180  ::close(pipes[0]);
181  }
182 
183  // Run the child hooks.
184  foreach (const Subprocess::ChildHook& hook, child_hooks) {
185  Try<Nothing> callback = hook();
186 
187  // If the callback failed, we should abort execution.
188  if (callback.isError()) {
189  ABORT("Failed to execute Subprocess::ChildHook: " + callback.error());
190  }
191  }
192 
193  os::execvpe(path.c_str(), argv, envp);
194 
195  ABORT("Failed to os::execvpe on path '" + path + "': " + os::strerror(errno));
196 }

199 inline Try<pid_t> cloneChild(
200  const std::string& path,
201  std::vector<std::string> argv,
202  const Option<std::map<std::string, std::string>>& environment,
203  const Option<lambda::function<
204  pid_t(const lambda::function<int()>&)>>& _clone,
205  const std::vector<Subprocess::ParentHook>& parent_hooks,
206  const std::vector<Subprocess::ChildHook>& child_hooks,
207  const InputFileDescriptors stdinfds,
208  const OutputFileDescriptors stdoutfds,
209  const OutputFileDescriptors stderrfds)
210 {
211  // The real arguments that will be passed to 'os::execvpe'. We need
212  // to construct them here before doing the clone as it might not be
213  // async signal safe to perform the memory allocation.
214  char** _argv = new char*[argv.size() + 1];
215  for (size_t i = 0; i < argv.size(); i++) {
216  _argv[i] = (char*) argv[i].c_str();
217  }
218  _argv[argv.size()] = nullptr;
219 
220  // Like above, we need to construct the environment that we'll pass
221  // to 'os::execvpe' as it might not be async-safe to perform the
222  // memory allocations.
223  char** envp = os::raw::environment();
224 
225  if (environment.isSome()) {
226  // NOTE: We add 1 to the size for a `nullptr` terminator.
227  envp = new char*[environment.get().size() + 1];
228 
229  size_t index = 0;
230  foreachpair (
231  const std::string& key,
232  const std::string& value, environment.get()) {
233  std::string entry = key + "=" + value;
234  envp[index] = new char[entry.size() + 1];
235  strncpy(envp[index], entry.c_str(), entry.size() + 1);
236  ++index;
237  }
238 
239  envp[index] = nullptr;
240  }
241 
242  // Determine the function to clone the child process. If the user
243  // does not specify the clone function, we will use the default.
244  lambda::function<pid_t(const lambda::function<int()>&)> clone =
245  (_clone.isSome() ? _clone.get() : defaultClone);
246 
247  // Currently we will block the child's execution of the new process
248  // until all the `parent_hooks` (if any) have executed.
249  std::array<int, 2> pipes;
250  const bool blocking = !parent_hooks.empty();
251 
252  if (blocking) {
253  // We assume this should not fail under reasonable conditions so we
254  // use CHECK.
255  Try<std::array<int, 2>> pipe = os::pipe();
256  CHECK_SOME(pipe);
257 
258  pipes = pipe.get();
259  }
260 
261  // Now, clone the child process.
262  pid_t pid = clone(lambda::bind(
263  &childMain,
264  path,
265  _argv,
266  envp,
267  stdinfds,
268  stdoutfds,
269  stderrfds,
270  blocking,
271  pipes.data(),
272  child_hooks));
273 
274  delete[] _argv;
275 
276  // Need to delete 'envp' if we had environment variables passed to
277  // us and we needed to allocate the space.
278  if (environment.isSome()) {
279  CHECK_NE(os::raw::environment(), envp);
280 
281  // We ignore the last 'envp' entry since it is nullptr.
282  for (size_t index = 0; index < environment->size(); index++) {
283  delete[] envp[index];
284  }
285 
286  delete[] envp;
287  }
288 
289  if (pid == -1) {
290  // Save the errno as 'close' below might overwrite it.
291  ErrnoError error("Failed to clone");
292  internal::close(stdinfds, stdoutfds, stderrfds);
293 
294  if (blocking) {
295  os::close(pipes[0]);
296  os::close(pipes[1]);
297  }
298 
299  return error;
300  }
301 
302  // Close the child-ends of the file descriptors that are created by
303  // this function.
304  internal::close({stdinfds.read, stdoutfds.write, stderrfds.write});
305 
306  if (blocking) {
307  os::close(pipes[0]);
308 
309  // Run the parent hooks.
310  foreach (const Subprocess::ParentHook& hook, parent_hooks) {
311  Try<Nothing> parentSetup = hook.parent_setup(pid);
312 
313  // If the hook callback fails, we shouldn't proceed with the
314  // execution and hence the child process should be killed.
315  if (parentSetup.isError()) {
316  LOG(WARNING)
317  << "Failed to execute Subprocess::ParentHook in parent for child '"
318  << pid << "': " << parentSetup.error();
319 
320  os::close(pipes[1]);
321 
322  // Ensure the child is killed.
323  ::kill(pid, SIGKILL);
324 
325  return Error(
326  "Failed to execute Subprocess::ParentHook in parent for child '" +
327  stringify(pid) + "': " + parentSetup.error());
328  }
329  }
330 
331  // Now that we've executed the parent hooks, we can signal the child to
332  // continue by writing to the pipe.
333  char dummy;
334  ssize_t length;
335  while ((length = ::write(pipes[1], &dummy, sizeof(dummy))) == -1 &&
336  errno == EINTR);
337 
338  os::close(pipes[1]);
339 
340  if (length != sizeof(dummy)) {
341  // Ensure the child is killed.
342  ::kill(pid, SIGKILL);
343 
344  return Error("Failed to synchronize child process");
345  }
346  }
347 
348  return pid;
349 }
