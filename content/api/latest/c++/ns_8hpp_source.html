153 // Re-associate the calling process with the specified namespace. The
154 // path refers to one of the corresponding namespace entries in the
155 // /proc/[pid]/ns/ directory (or bind mounted elsewhere). We do not
156 // allow a process with multiple threads to call this function because
157 // it will lead to some weird situations where different threads of a
158 // process are in different namespaces.
159 inline Try<Nothing> setns(
160  const std::string& path,
161  const std::string& ns,
162  bool checkMultithreaded = true)
163 {
164  if (checkMultithreaded) {
165  // Return error if there're multiple threads in the calling process.
166  Try<std::set<pid_t>> threads = proc::threads(::getpid());
167  if (threads.isError()) {
168  return Error(
169  "Failed to get the threads of the current process: " +
170  threads.error());
171  } else if (threads.get().size() > 1) {
172  return Error("Multiple threads exist in the current process");
173  }
174  }
175 
176  if (ns::namespaces().count(ns) == 0) {
177  return Error("Namespace '" + ns + "' is not supported");
178  }
179 
180  // Currently, we don't support pid namespace as its semantics is
181  // different from other namespaces (instead of re-associating the
182  // calling thread, it re-associates the *children* of the calling
183  // thread with the specified namespace).
184  if (ns == "pid") {
185  return Error("Pid namespace is not supported");
186  }
187 
188  Try<int> fd = os::open(path, O_RDONLY | O_CLOEXEC);
189 
190  if (fd.isError()) {
191  return Error("Failed to open '" + path + "': " + fd.error());
192  }
193 
194  Try<int> nstype = ns::nstype(ns);
195  if (nstype.isError()) {
196  return Error(nstype.error());
197  }
198 
199  if (::setns(fd.get(), nstype.get()) == -1) {
200  // Save the errno as it might be overwritten by 'os::close' below.
201  ErrnoError error;
202  os::close(fd.get());
203  return error;
204  }
205 
206  os::close(fd.get());
207  return Nothing();
208 }

211 // Re-associate the calling process with the specified namespace. The
212 // pid specifies the process whose namespace we will associate.
213 inline Try<Nothing> setns(pid_t pid, const std::string& ns)
214 {
215  if (!os::exists(pid)) {
216  return Error("Pid " + stringify(pid) + " does not exist");
217  }
218 
219  std::string path = path::join("/proc", stringify(pid), "ns", ns);
220  if (!os::exists(path)) {
221  return Error("Namespace '" + ns + "' is not supported");
222  }
223 
224  return ns::setns(path, ns);
225 }

228 // Get the inode number of the specified namespace for the specified
229 // pid. The inode number identifies the namespace and can be used for
230 // comparisons, i.e., two processes with the same inode for a given
231 // namespace type are in the same namespace.
232 inline Try<ino_t> getns(pid_t pid, const std::string& ns)
233 {
234  if (!os::exists(pid)) {
235  return Error("Pid " + stringify(pid) + " does not exist");
236  }
237 
238  if (ns::namespaces().count(ns) < 1) {
239  return Error("Namespace '" + ns + "' is not supported");
240  }
241 
242  std::string path = path::join("/proc", stringify(pid), "ns", ns);
243  struct stat s;
244  if (::stat(path.c_str(), &s) < 0) {
245  return ErrnoError("Failed to stat " + ns + " namespace handle"
246  " for pid " + stringify(pid));
247  }
248 
249  return s.st_ino;
250 }

292 inline Try<pid_t> clone(
293  pid_t target,
294  int nstypes,
295  const lambda::function<int()>& f,
296  int flags)
297 {
298  // NOTE: the order in which we 'setns' is significant, so we use an
299  // array here rather than something like a map.
300  //
301  // The user namespace needs to be entered first if we need to
302  // increase the privilege and last if we want to decrease the
303  // privilege. Said another way, entering the user namespace first
304  // gives an unprivileged user the potential to enter the other
305  // namespaces.
306  const size_t NAMESPACES = 7;
307  struct
308  {
309  int nstype;
310  std::string name;
311  } namespaces[NAMESPACES] = {
312  {CLONE_NEWUSER, "user"},
313  {CLONE_NEWCGROUP, "cgroup"},
314  {CLONE_NEWIPC, "ipc"},
315  {CLONE_NEWUTS, "uts"},
316  {CLONE_NEWNET, "net"},
317  {CLONE_NEWPID, "pid"},
318  {CLONE_NEWNS, "mnt"}
319  };
320 
321  // Support for user namespaces in all filesystems is incomplete
322  // until version 3.12 (see 'Availability' in man page of
323  // 'user_namespaces'), so for now we don't support entering them.
324  //
325  // TODO(benh): Support user namespaces if the current system can
326  // support it, e.g., check the kernel version number or try and do a
327  // clone with CLONE_NEWUSER to see if it works. NOTE: before we can
328  // fully support user namespaces, however, we must take care to
329  // either enter the user namespace first or last. We'll want to
330  // enter it first if we need to increase the privilege and last if
331  // we want to decrease the privilege. Currently nsenter.c from
332  // utils-linux does this via doing two passes to make sure we either
333  // enter first or last. We'll need to do something similar here once
334  // we support user namespaces as well.
335  if (nstypes & CLONE_NEWUSER) {
336  return Error("User namespaces are not supported");
337  }
338 
339  // File descriptors keyed by the (parent) namespace we are entering.
340  hashmap<int, int> fds = {};
341 
342  // Helper for closing a list of file descriptors.
343  auto close = [](const std::list<int>& fds) {
344  foreach (int fd, fds) {
345  ::close(fd); // Need to call the async-signal safe version.
346  }
347  };
348 
349  // NOTE: we do all of this ahead of time so we can be async signal
350  // safe after calling fork below.
351  for (size_t i = 0; i < NAMESPACES; i++) {
352  // Only open the namespace file descriptor if it's been requested.
353  if (namespaces[i].nstype & nstypes) {
354  std::string path =
355  path::join("/proc", stringify(target), "ns", namespaces[i].name);
356  Try<int> fd = os::open(path, O_RDONLY);
357  if (fd.isError()) {
358  close(fds.values());
359  return Error("Failed to open '" + path +
360  "' for entering namespace: " + fd.error());
361  }
362  fds[namespaces[i].nstype] = fd.get();
363  }
364  }
365 
366  // We use a domain socket rather than pipes so that we can send back
367  // the PID of the final child process. The parent socket is
368  // `sockets[0]` and the child socket is `sockets[1]`. Note that both
369  // sockets are both read/write but currently only the parent reads
370  // and the child writes.
371  int sockets[2] = {-1, -1};
372  if (socketpair(AF_UNIX, SOCK_STREAM, 0, sockets) != 0) {
373  close(fds.values());
374  return ErrnoError("Failed to create Unix domain socket");
375  }
376 
377  // Need to set SO_PASSCRED option in order to receive credentials
378  // (which is how we get the pid of the clone'd process, see
379  // below). Note that apparently we only need to do this for
380  // receiving, not also for sending.
381  const int value = 1;
382  const ssize_t size = sizeof(value);
383  if (setsockopt(sockets[0], SOL_SOCKET, SO_PASSCRED, &value, size) == -1) {
384  Error error = ErrnoError("Failed to set socket option SO_PASSCRED");
385  close(fds.values());
386  ::close(sockets[0]);
387  ::close(sockets[1]);
388  return error;
389  }
390 
391  // NOTE: to determine the pid of the final process executing the
392  // specified lambda we use the SCM_CREDENTIALS mechanism of
393  // 'sendmsg' and 'recvmsg'. On Linux there is also a way to do this
394  // via 'getsockopt' and SO_PEERCRED which looks easier, but IIUC
395  // requires you to do an explicit connect from the child process
396  // back to the parent so that there is only one connection per
397  // socket (unlike in our world where the socket can be used by
398  // multiple forks/clones simultaneously because it's just a file
399  // descriptor that gets copied after each fork/clone). Perhaps the
400  // SO_PEERCRED is less lines of code but this approach was taken for
401  // now.
402 
403  char base[1];
404 
405  iovec iov = {0};
406  iov.iov_base = base;
407  iov.iov_len = sizeof(base);
408 
409  // Need to allocate a char array large enough to hold "control"
410  // data. However, since this buffer is in reality a 'struct cmsghdr'
411  // we use a union to ensure that it is aligned as required for that
412  // structure.
413  union {
414  struct cmsghdr cmessage;
415  char control[CMSG_SPACE(sizeof(struct ucred))];
416  };
417 
418  cmessage.cmsg_len = CMSG_LEN(sizeof(struct ucred));
419  cmessage.cmsg_level = SOL_SOCKET;
420  cmessage.cmsg_type = SCM_CREDENTIALS;
421 
422  msghdr message = {0};
423  message.msg_name = nullptr;
424  message.msg_namelen = 0;
425  message.msg_iov = &iov;
426  message.msg_iovlen = 1;
427  message.msg_control = control;
428  message.msg_controllen = sizeof(control); // CMSG_LEN(sizeof(struct ucred));
429 
430  // Finally, the stack we'll use in the call to os::clone below (we
431  // allocate the stack here in order to keep the call to os::clone
432  // async signal safe, since otherwise it would be doing the dynamic
433  // allocation itself).
434  Try<os::Stack> stack = os::Stack::create(os::Stack::DEFAULT_SIZE);
435  if (stack.isError()) {
436  return Error("Failed to allocate stack: " + stack.error());
437  }
438 
439  pid_t child = fork();
440  if (child < 0) {
441  stack->deallocate();
442  close(fds.values());
443  ::close(sockets[0]);
444  ::close(sockets[1]);
445  return ErrnoError();
446  } else if (child > 0) {
447  // Parent.
448  stack->deallocate();
449 
450  close(fds.values());
451  ::close(sockets[1]);
452 
453  ssize_t length = recvmsg(sockets[0], &message, 0);
454 
455  // TODO(benh): Note that whenever we 'kill(child, SIGKILL)' below
456  // we don't guarantee cleanup! It's possible that the
457  // greatgrandchild is still running. Require the greatgrandchild
458  // to read from the socket after sending back it's pid to ensure
459  // no orphans.
460 
461  if (length < 0) {
462  // We failed to read, close the socket and kill the child
463  // (which might die on it's own trying to write to the
464  // socket).
465  Error error = ErrnoError("Failed to receive");
466  ::close(sockets[0]);
467  kill(child, SIGKILL);
468  return error;
469  } else if (length == 0) {
470  // Socket closed, child must have died, but kill anyway.
471  ::close(sockets[0]);
472  kill(child, SIGKILL);
473  return Error("Failed to receive: Socket closed");
474  }
475 
476  ::close(sockets[0]);
477 
478  // Extract pid.
479  if (CMSG_FIRSTHDR(&message) == nullptr ||
480  CMSG_FIRSTHDR(&message)->cmsg_len != CMSG_LEN(sizeof(struct ucred)) ||
481  CMSG_FIRSTHDR(&message)->cmsg_level != SOL_SOCKET ||
482  CMSG_FIRSTHDR(&message)->cmsg_type != SCM_CREDENTIALS) {
483  kill(child, SIGKILL);
484  return Error("Bad control data received");
485  }
486 
487  pid_t pid = ((struct ucred*) CMSG_DATA(CMSG_FIRSTHDR(&message)))->pid;
488 
489  // Need to `waitpid` on child process to avoid a zombie. Note that
490  // it's expected that the child will terminate quickly hence
491  // blocking here.
492  int status;
493  while (true) {
494  if (waitpid(child, &status, 0) == -1) {
495  if (errno == EINTR) {
496  continue;
497  } else {
498  return ErrnoError("Failed to `waitpid` on child");
499  }
500  } else if (WIFSTOPPED(status)) {
501  continue;
502  } else {
503  break;
504  }
505  }
506 
507  CHECK(WIFEXITED(status) || WIFSIGNALED(status))
508  << "Unexpected wait status " << status;
509 
510  if (!WSUCCEEDED(status)) {
511  return Error("Failed to clone: " + WSTRINGIFY(status));
512  }
513 
514  return pid;
515  } else {
516  // Child.
517  ::close(sockets[0]);
518 
519  // Loop through and 'setns' into all of the parent namespaces that
520  // have been requested.
521  for (size_t i = 0; i < NAMESPACES; i++) {
522  Option<int> fd = fds.get(namespaces[i].nstype);
523  if (fd.isSome()) {
524  assert(namespaces[i].nstype & nstypes);
525  if (::setns(fd.get(), namespaces[i].nstype) < 0) {
526  close(fds.values());
527  ::close(sockets[1]);
528  _exit(EXIT_FAILURE);
529  }
530  }
531  }
532 
533  close(fds.values());
534 
535  // Fork again to make sure we're actually in those namespaces
536  // (required for the pid namespace at least).
537  //
538  // TODO(benh): Don't do a fork if we're not actually entering the
539  // PID namespace since the extra fork is unnecessary.
540  pid_t grandchild = fork();
541  if (grandchild < 0) {
542  // TODO(benh): Exit with `errno` in order to capture `fork` error?
543  ::close(sockets[1]);
544  _exit(EXIT_FAILURE);
545  } else if (grandchild > 0) {
546  // Still the (first) child.
547  ::close(sockets[1]);
548 
549  // Need to reap the grandchild and then just exit since we're no
550  // longer necessary. Technically when the grandchild exits it'll
551  // be reaped but by doing a `waitpid` we can better propagate
552  // back any errors that might have occurred with the grandchild.
553  int status;
554  while (true) {
555  if (waitpid(grandchild, &status, 0) == -1) {
556  if (errno == EINTR) {
557  continue;
558  } else {
559  _exit(1);
560  }
561  } else if (WIFSTOPPED(status)) {
562  continue;
563  } else {
564  break;
565  }
566  }
567 
568  assert(WIFEXITED(status) || WIFSIGNALED(status));
569 
570  if (WIFEXITED(status)) {
571  _exit(WEXITSTATUS(status));
572  }
573 
574  assert(WIFSIGNALED(status));
575  raise(WTERMSIG(status));
576  }
577 
578  // Grandchild (second child, now completely entered in the
579  // namespaces of the target).
580  //
581  // Now clone with the specified flags, close the unused socket,
582  // and execute the specified function.
583  pid_t pid = os::clone([=]() {
584  // Now send back the pid and have it be translated appropriately
585  // by the kernel to the enclosing pid namespace.
586  //
587  // NOTE: sending back the pid is best effort because we're going
588  // to exit no matter what.
589  ((struct ucred*) CMSG_DATA(CMSG_FIRSTHDR(&message)))->pid = ::getpid();
590  ((struct ucred*) CMSG_DATA(CMSG_FIRSTHDR(&message)))->uid = ::getuid();
591  ((struct ucred*) CMSG_DATA(CMSG_FIRSTHDR(&message)))->gid = ::getgid();
592 
593  if (sendmsg(sockets[1], &message, 0) == -1) {
594  // Failed to send the pid back to the parent!
595  _exit(EXIT_FAILURE);
596  }
597 
598  ::close(sockets[1]);
599 
600  return f();
601  },
602  flags,
603  stack.get());
604 
605  ::close(sockets[1]);
606 
607  // TODO(benh): Kill ourselves with an exit status that we can
608  // decode above to determine why `clone` failed.
609  _exit(pid < 0 ? EXIT_FAILURE : EXIT_SUCCESS);
610  }
611  UNREACHABLE();
612 }

617 inline process::Future<Nothing> destroy(ino_t inode)
618 {
619  // Check we're not trying to kill the root namespace.
620  Try<ino_t> ns = ns::getns(1, "pid");
621  if (ns.isError()) {
622  return process::Failure(ns.error());
623  }
624 
625  if (ns.get() == inode) {
626  return process::Failure("Cannot destroy root pid namespace");
627  }
628 
629  // Or ourselves.
630  ns = ns::getns(::getpid(), "pid");
631  if (ns.isError()) {
632  return process::Failure(ns.error());
633  }
634 
635  if (ns.get() == inode) {
636  return process::Failure("Cannot destroy own pid namespace");
637  }
638 
639  // Signal all pids in the namespace, including the init pid if it's
640  // still running. Once the init pid has been signalled the kernel
641  // will prevent any new children forking in the namespace and will
642  // also signal all other pids in the namespace.
643  Try<std::set<pid_t>> pids = os::pids();
644  if (pids.isError()) {
645  return process::Failure("Failed to list of processes");
646  }
647 
648  foreach (pid_t pid, pids.get()) {
649  // Ignore any errors, probably because the process no longer
650  // exists, and ignorable otherwise.
651  Try<ino_t> ns = ns::getns(pid, "pid");
652  if (ns.isSome() && ns.get() == inode) {
653  kill(pid, SIGKILL);
654  }
655  }
656 
657  // Get a new snapshot and do a second pass of the pids to capture
658  // any pids that are dying so we can reap them.
659  pids = os::pids();
660  if (pids.isError()) {
661  return process::Failure("Failed to list of processes");
662  }
663 
664  std::list<process::Future<Option<int>>> futures;
665 
666  foreach (pid_t pid, pids.get()) {
667  Try<ino_t> ns = ns::getns(pid, "pid");
668  if (ns.isSome() && ns.get() == inode) {
669  futures.push_back(process::reap(pid));
670  }
671 
672  // Ignore any errors, probably because the process no longer
673  // exists, and ignorable otherwise.
674  }
675 
676  // Wait for all the signalled processes to terminate. The pid
677  // namespace will then be empty and will be released by the kernel
678  // (unless there are additional references).
679  return process::collect(futures)
680  .then([]() { return Nothing(); });
681 }

686 // Returns the namespace flags in the string form of bitwise-ORing the
687 // flags, e.g., CLONE_NEWNS | CLONE_NEWNET.
688 inline std::string stringify(int flags)
689 {
690  hashmap<unsigned int, std::string> names = {
691  {CLONE_NEWNS, "CLONE_NEWNS"},
692  {CLONE_NEWUTS, "CLONE_NEWUTS"},
693  {CLONE_NEWIPC, "CLONE_NEWIPC"},
694  {CLONE_NEWPID, "CLONE_NEWPID"},
695  {CLONE_NEWNET, "CLONE_NEWNET"},
696  {CLONE_NEWUSER, "CLONE_NEWUSER"},
697  {CLONE_NEWCGROUP, "CLONE_NEWCGROUP"}
698  };
699 
700  std::vector<std::string> namespaces;
701  foreachpair (unsigned int flag, const std::string& name, names) {
702  if (flags & flag) {
703  namespaces.push_back(name);
704  }
705  }
706 
707  return strings::join(" | ", namespaces);
708 }
