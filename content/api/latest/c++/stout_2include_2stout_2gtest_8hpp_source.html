220 // NOTE: On Windows, the closest equivalent to `sleep` is `timeout`.
221 // Unfortunately, `timeout` requires an interactive terminal, otherwise
222 // it errors out immediately. Instead, we use `ping` against localhost
223 // with a count. On Windows, `ping` waits one second between pings.
224 // Additionally, because `ping` requires a count greater than 0,
225 // we simply `exit 0` if the sleep is too short.
226 #ifndef __WINDOWS__
227 #define SLEEP_COMMAND(x) "sleep " #x
228 #else
229 #define SLEEP_COMMAND(x) (x > 0 ? "ping 127.0.0.1 -n " #x : "cmd /C exit 0")

240 inline ::testing::AssertionResult AssertExited(
241  const char* actualExpr,
242  const int actual)
243 {
244  if (WIFEXITED(actual)) {
245  return ::testing::AssertionSuccess();
246  } else if (WIFSIGNALED(actual)) {
247  return ::testing::AssertionFailure()
248  << "Expecting WIFEXITED(" << actualExpr << ") but "
249  << " WIFSIGNALED(" << actualExpr << ") is true and "
250  << "WTERMSIG(" << actualExpr << ") is " << strsignal(WTERMSIG(actual));
251  } else if (WIFSTOPPED(actual)) {
252  return ::testing::AssertionFailure()
253  << "Expecting WIFEXITED(" << actualExpr << ") but"
254  << " WIFSTOPPED(" << actualExpr << ") is true and "
255  << "WSTOPSIG(" << actualExpr << ") is " << strsignal(WSTOPSIG(actual));
256  }
257 
258  return ::testing::AssertionFailure()
259  << "Expecting WIFEXITED(" << actualExpr << ") but got"
260  << " unknown value: " << ::testing::PrintToString(actual);
261 }
