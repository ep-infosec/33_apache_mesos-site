939  const std::string& weightsFlag)
940 {

942  std::vector<std::string> tokens = strings::tokenize(weightsFlag, ",");
943  foreach (const std::string& token, tokens) {
944  std::vector<std::string> pair = strings::tokenize(token, "=");
945  EXPECT_EQ(2u, pair.size());
946  double weight = atof(pair[1].c_str());
947  TWeightInfo weightInfo;
948  weightInfo.set_role(pair[0]);
949  weightInfo.set_weight(weight);
950  infos.Add()->CopyFrom(weightInfo);
951  }
952 
953  return infos;
