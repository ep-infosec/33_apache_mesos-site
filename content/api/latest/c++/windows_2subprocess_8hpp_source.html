40 // Retrieves system environment in a `std::map`, ignoring
41 // the current process's environment variables.
42 inline Option<std::map<std::string, std::string>> getSystemEnvironment()
43 {
44  std::wstring_convert<std::codecvt<wchar_t, char, mbstate_t>,
45  wchar_t> converter;
46 
47  std::map<std::string, std::string> systemEnvironment;
48  wchar_t* environmentEntry = nullptr;
49 
50  // Get the system environment.
51  // The third parameter (bool) tells the function *not* to inherit
52  // variables from the current process.
53  if (!CreateEnvironmentBlock((LPVOID*)&environmentEntry, nullptr, FALSE)) {
54  return None();
55  }
56 
57  // Save the environment block in order to destroy it later.
58  wchar_t* environmentBlock = environmentEntry;
59 
60  while (*environmentEntry != L'\0') {
61  // Each environment block contains the environment variables as follows:
62  // Var1=Value1\0
63  // Var2=Value2\0
64  // Var3=Value3\0
65  // ...
66  // VarN=ValueN\0\0
67  // The name of an environment variable cannot include an equal sign (=).
68 
69  wchar_t * separator = wcschr(environmentEntry, L'=');
70  std::wstring varName = std::wstring(environmentEntry, separator);
71  std::wstring varVal = std::wstring(separator + 1);
72 
73  // Mesos variables are upper case. Convert system variables to
74  // match the name provided by the scheduler in case of a collision.
75  std::transform(varName.begin(), varName.end(), varName.begin(), ::towupper);
76 
77  // The system environment has priority. Force `ANSI` usage until the code
78  // is converted to UNICODE.
79  systemEnvironment.insert_or_assign(
80  converter.to_bytes(varName.c_str()),
81  converter.to_bytes(varVal.c_str()));
82 
83  environmentEntry += varName.length() + varVal.length() + 2;
84  }
85 
86  DestroyEnvironmentBlock(environmentBlock);
87 
88  return systemEnvironment;
89 }

91 // Creates a null-terminated array of null-terminated strings that will be
92 // passed to `CreateProcess` as the `lpEnvironment` argument, as described by
93 // MSDN[1]. This array needs to be sorted in alphabetical order, but the `map`
94 // already takes care of that. Note that this function does not handle Unicode
95 // environments, so it should not be used in conjunction with the
96 // `CREATE_UNICODE_ENVIRONMENT` flag.
97 //
98 // NOTE: This function will add the system's environment variables into
99 // the returned string. These variables take precedence over the provided
100 // `env` and are generally necessary in order to launch things on Windows.
101 //
102 // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682425(v=vs.85).aspx
103 inline Option<std::string> createProcessEnvironment(
104  const Option<std::map<std::string, std::string>>& env)
105 {
106  if (env.isNone() || (env.isSome() && env.get().size() == 0)) {
107  return None();
108  }
109 
110  Option<std::map<std::string, std::string>> systemEnvironment =
111  getSystemEnvironment();
112 
113  // The system environment must be non-empty.
114  // No subprocesses will be able to launch if the system environment is blank.
115  CHECK(systemEnvironment.isSome() && systemEnvironment.get().size() > 0);
116 
117  std::map<std::string, std::string> combinedEnvironment = env.get();
118 
119  foreachpair (const std::string& key,
120  const std::string& value,
121  systemEnvironment.get()) {
122  combinedEnvironment[key] = value;
123  }
124 
125  std::string environmentString;
126  foreachpair (const std::string& key,
127  const std::string& value,
128  combinedEnvironment) {
129  environmentString += key + '=' + value + '\0';
130  }
131 
132  return environmentString;
133 }

136 inline Try<PROCESS_INFORMATION> createChildProcess(
137  const std::string& path,
138  const std::vector<std::string>& argv,
139  const Option<std::map<std::string, std::string>>& environment,
140  const InputFileDescriptors stdinfds,
141  const OutputFileDescriptors stdoutfds,
142  const OutputFileDescriptors stderrfds,
143  const std::vector<Subprocess::ParentHook>& parent_hooks)
144 {
145  // Construct the environment that will be passed to `CreateProcess`.
146  Option<std::string> environmentString = createProcessEnvironment(environment);
147  const char* processEnvironment = environmentString.isNone()
148  ? nullptr
149  : environmentString.get().c_str();
150 
151  PROCESS_INFORMATION processInfo;
152  STARTUPINFO startupInfo;
153 
154  ::ZeroMemory(&processInfo, sizeof(PROCESS_INFORMATION));
155  ::ZeroMemory(&startupInfo, sizeof(STARTUPINFO));
156 
157  // Hook up the `stdin`/`stdout`/`stderr` pipes and use the
158  // `STARTF_USESTDHANDLES` flag to instruct the child to use them[1]. A more
159  // user-friendly example can be found in [2].
160  //
161  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms686331(v=vs.85).aspx
162  // [2] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682499(v=vs.85).aspx
163  startupInfo.cb = sizeof(STARTUPINFO);
164  startupInfo.hStdInput = stdinfds.read;
165  startupInfo.hStdOutput = stdoutfds.write;
166  startupInfo.hStdError = stderrfds.write;
167  startupInfo.dwFlags |= STARTF_USESTDHANDLES;
168 
169  // Build command to pass to `::CreateProcess`.
170  //
171  // NOTE: We are expecting that components of `argv` that need to be quoted
172  // (for example, paths with spaces in them like `C:\"Program Files"\foo.exe`)
173  // to have been already quoted correctly before we generate `command`.
174  // Incorrectly-quoted command arguments will probably lead the child process
175  // to terminate with an error. See also NOTE on `process::subprocess`.
176  std::string command = strings::join(" ", argv);
177 
178  // Escape the quotes in `command`.
179  //
180  // TODO(dpravat): Add tests cases that cover this functionality. See
181  // MESOS-5418.
182  command = strings::replace(command, "\"", "\\\"");
183 
184  // NOTE: If Mesos is built against the ANSI version of this function, the
185  // environment is limited to 32,767 characters. See[1].
186  //
187  // TODO(hausdorff): Figure out how to map the `path` and `args` arguments of
188  // this function into a call to `::CreateProcess` that is more general
189  // purpose. In particular, on POSIX, we expect that calls to `subprocess` can
190  // be called with relative `path` (e.g., it could simply be `sh`). However,
191  // on Windows, unlike the calls to (e.g.) `exec`, `::CreateProcess` will
192  // expect that this argument be a fully-qualified path. In the end, we'd like
193  // the calls to `subprocess` to have similar command formats to minimize
194  // confusion and mistakes.
195  //
196  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682425(v=vs.85).aspx
197  BOOL createProcessResult = CreateProcess(
198  nullptr,
199  (LPSTR)command.data(),
200  nullptr, // Default security attributes.
201  nullptr, // Default primary thread security attributes.
202  TRUE, // Inherited parent process handles.
203  CREATE_SUSPENDED, // Create process in suspended state.
204  (LPVOID)processEnvironment,
205  nullptr, // Use parent's current directory.
206  &startupInfo, // STARTUPINFO pointer.
207  &processInfo); // PROCESS_INFORMATION pointer.
208 
209  if (!createProcessResult) {
210  return WindowsError(
211  "Failed to call CreateProcess on command '" + command + "'");
212  }
213 
214  // Run the parent hooks.
215  const pid_t pid = processInfo.dwProcessId;
216  foreach (const Subprocess::ParentHook& hook, parent_hooks) {
217  Try<Nothing> parentSetup = hook.parent_setup(pid);
218 
219  // If the hook callback fails, we shouldn't proceed with the
220  // execution and hence the child process should be killed.
221  if (parentSetup.isError()) {
222  // Attempt to kill the process. Since it is still in suspended state, we
223  // do not need to kill any descendents. We also can't use `os::kill_job`
224  // because this process is not in a Job Object unless one of the parent
225  // hooks added it.
226  ::TerminateProcess(processInfo.hProcess, 1);
227 
228  return Error(
229  "Failed to execute Parent Hook in child '" + stringify(pid) +
230  "' with command '" + command + "': " + parentSetup.error());
231  }
232  }
233 
234  // Start child process.
235  if (::ResumeThread(processInfo.hThread) == -1) {
236  return WindowsError(
237  "Failed to resume child process with command '" + command + "'");
238  }
239 
240  return processInfo;
241 }
