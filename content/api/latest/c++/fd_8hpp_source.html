30 // The `WindowsFD` class exists to provide an common interface with the POSIX
31 // file descriptor. While the bare `int` representation of the POSIX file
32 // descriptor API is undesirable, we rendezvous there in order to maintain the
33 // existing code in Mesos.
34 //
35 // In the platform-agnostic code paths, the `int_fd` type is aliased to
36 // `WindowsFD`. The `os::*` functions return a type appropriate to the platform,
37 // which allows us to write code like this:
38 //
39 // Try<int_fd> fd = os::open(...);
40 //
41 // The `WindowsFD` constructs off one of:
42 // (1) `int` - from the WinCRT API
43 // (2) `HANDLE` - from the Win32 API
44 // (3) `SOCKET` - from the WinSock API
45 //
46 // The `os::*` functions then take an instance of `WindowsFD`, examines
47 // the state and dispatches to the appropriate API.
48

68  // IMPORTANT: The `HANDLE` here is expected to be file handles. Specifically,
69  // `HANDLE`s returned by file API such as `CreateFile`. There are
70  // APIs that return `HANDLE`s with different error values, and
71  // therefore must be handled accordingly. For example, a thread API
72  // such as `CreateThread` returns `NULL` as the error value, rather
73  // than `INVALID_HANDLE_VALUE`.
74  // TODO(mpark): Consider adding a second parameter which tells us what the
75  // error values are.
76  WindowsFD(HANDLE handle)
77  : type_(FD_HANDLE),
78  crt_(
79  handle == INVALID_HANDLE_VALUE
80  ? -1
81  : ::_open_osfhandle(reinterpret_cast<intptr_t>(handle), O_RDWR)),
82  handle_(handle) {}

124  union

126  // We keep both a CRT FD as well as a `HANDLE`
127  // regardless of whether we were constructed
128  // from a file or a handle.
129  //
130  // This is because once we request for a CRT FD
131  // from a `HANDLE`, we're required to close it
132  // via `_close`. If we were to do the conversion
133  // lazily upon request, the resulting CRT FD

165 // The complexity in this function is due to our effort in trying to support the
166 // cases where file descriptors are compared as an `int` on POSIX. For example,
167 // we use expressions such as `fd < 0` to check for validity.
168 // TODO(mpark): Consider introducing an `is_valid` function for `int_fd`.
169 inline bool operator<(const WindowsFD& left, const WindowsFD& right)
170 {
171  // In general, when compared against a `WindowsFD` in the `FD_CRT`, we map
172  // `INVALID_HANDLE_VALUE` and `INVALID_SOCKET` to `-1` before performing the
173  // comparison. The check for `< 0` followed by cast to `HANDLE` or `SOCKET` is
174  // due to the fact that `HANDLE` and `SOCKET` are both unsigned.
175  switch (left.type()) {
176  case WindowsFD::FD_CRT: {
177  switch (right.type()) {
178  case WindowsFD::FD_CRT: {
179  return left.crt() < right.crt();
180  }
181  case WindowsFD::FD_HANDLE: {
182  if (static_cast<HANDLE>(right) == INVALID_HANDLE_VALUE) {
183  return left.crt() < -1;
184  }
185  if (left.crt() < 0) {
186  return true;
187  }
188 #pragma warning(push)
189 #pragma warning(disable : 4312)
190  // Disable `int`-to-`HANDLE` compiler warning. This is safe to do,
191  // see comment above.
192  return reinterpret_cast<HANDLE>(left.crt()) <
193  static_cast<HANDLE>(right);
194 #pragma warning(pop)
195  }
196  case WindowsFD::FD_SOCKET: {
197  if (static_cast<SOCKET>(right) == INVALID_SOCKET) {
198  return left.crt() < -1;
199  }
200  if (left.crt() < 0) {
201  return true;
202  }
203  return static_cast<SOCKET>(left.crt()) < static_cast<SOCKET>(right);
204  }
205  }
206  }
207  case WindowsFD::FD_HANDLE: {
208  switch (right.type()) {
209  case WindowsFD::FD_CRT: {
210  if (static_cast<HANDLE>(left) == INVALID_HANDLE_VALUE) {
211  return -1 < right.crt();
212  }
213  if (right.crt() < 0) {
214  return false;
215  }
216 #pragma warning(push)
217 #pragma warning(disable : 4312)
218  // Disable `int`-to-`HANDLE` compiler warning. This is safe to do,
219  // see comment above.
220  return static_cast<HANDLE>(left) <
221  reinterpret_cast<HANDLE>(right.crt());
222 #pragma warning(pop)
223  }
224  case WindowsFD::FD_HANDLE: {
225  return static_cast<HANDLE>(left) < static_cast<HANDLE>(right);
226  }
227  case WindowsFD::FD_SOCKET: {
228  return static_cast<HANDLE>(left) <
229  reinterpret_cast<HANDLE>(static_cast<SOCKET>(right));
230  }
231  }
232  }
233  case WindowsFD::FD_SOCKET: {
234  switch (right.type()) {
235  case WindowsFD::FD_CRT: {
236  if (static_cast<SOCKET>(left) == INVALID_SOCKET) {
237  return -1 < right.crt();
238  }
239  if (right.crt() < 0) {
240  return false;
241  }
242  return static_cast<SOCKET>(left) < static_cast<SOCKET>(right.crt());
243  }
244  case WindowsFD::FD_HANDLE: {
245  return reinterpret_cast<HANDLE>(static_cast<SOCKET>(left)) <
246  static_cast<HANDLE>(right);
247  }
248  case WindowsFD::FD_SOCKET: {
249  return static_cast<SOCKET>(left) < static_cast<SOCKET>(right);
250  }
251  }
252  }
253  }
254  UNREACHABLE();
255 }
