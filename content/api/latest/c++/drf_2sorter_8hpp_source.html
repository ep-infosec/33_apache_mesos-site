156  // Total resources.
157  struct Total {
158  // We need to keep track of the resources (and not just scalar quantities)
159  // to account for multiple copies of the same shared resources. We need to
160  // ensure that we do not update the scalar quantities for shared resources
161  // when the change is only in the number of copies in the sorter.
162  hashmap<SlaveID, Resources> resources;
163 
164  // NOTE: Scalars can be safely aggregated across slaves. We keep
165  // that to speed up the calculation of shares. See MESOS-2891 for
166  // the reasons why we want to do that.
167  //
168  // NOTE: We omit information about dynamic reservations and persistent
169  // volumes here to enable resources to be aggregated across slaves
170  // more effectively. See MESOS-4833 for more information.
171  //
172  // Sharedness info is also stripped out when resource identities are
173  // omitted because sharedness inherently refers to the identities of
174  // resources and not quantities.
175  Resources scalarQuantities;
176 
177  // We also store a map version of `scalarQuantities`, mapping
178  // the `Resource::name` to aggregated scalar. This improves the
179  // performance of calculating shares. See MESOS-4694.
180  //
181  // TODO(bmahler): Ideally we do not store `scalarQuantities`
182  // redundantly here, investigate performance improvements to
183  // `Resources` to make this unnecessary.
184  hashmap<std::string, Value::Scalar> totals;
185  } total_;

187  // Allocation for a client.
188  struct Allocation {
189  // We maintain multiple copies of each shared resource allocated

191  // of times this shared resource has been allocated to (and has
192  // not been recovered from) a specific client.
193  hashmap<SlaveID, Resources> resources;

195  // Similarly, we aggregate scalars across slaves and omit information
196  // about dynamic reservations, persistent volumes and sharedness of
197  // the corresponding resource. See notes above.
