78 // Deserializes a string message into a protobuf message based on the
79 // HTTP content type.
80 template <typename Message>
81 Try<Message> deserialize(
82  ContentType contentType,
83  const std::string& body)
84 {
85  switch (contentType) {
86  case ContentType::PROTOBUF: {
87  Message message;
88  if (!message.ParseFromString(body)) {
89  return Error("Failed to parse body into a protobuf object");
90  }
91  return message;
92  }
93  case ContentType::JSON: {
94  Try<JSON::Value> value = JSON::parse(body);
95  if (value.isError()) {
96  return Error("Failed to parse body into JSON: " + value.error());
97  }
98 
99  return ::protobuf::parse<Message>(value.get());
100  }
101  case ContentType::RECORDIO: {
102  return Error("Deserializing a RecordIO stream is not supported");
103  }
104  }
105 
106  UNREACHABLE();
107 }
