29 // Initializes the _global_ OpenSSL context (SSL_CTX) as well as the
30 // crypto library in order to support multi-threading. The global
31 // context gets initialized using the environment variables:
32 //
33 // LIBPROCESS_SSL_ENABLED=(false|0,true|1)
34 // LIBPROCESS_SSL_SUPPORT_DOWNGRADE=(false|0,true|1)
35 // LIBPROCESS_SSL_CERT_FILE=(path to certificate)
36 // LIBPROCESS_SSL_KEY_FILE=(path to key)
37 // LIBPROCESS_SSL_VERIFY_CERT=(false|0,true|1)
38 // LIBPROCESS_SSL_REQUIRE_CERT=(false|0,true|1)
39 // LIBPROCESS_SSL_VERIFY_IPADD=(false|0,true|1)
40 // LIBPROCESS_SSL_VERIFY_DEPTH=(4)
41 // LIBPROCESS_SSL_CA_DIR=(path to CA directory)
42 // LIBPROCESS_SSL_CA_FILE=(path to CA file)
43 // LIBPROCESS_SSL_CIPHERS=(accepted ciphers separated by ':')
44 // LIBPROCESS_SSL_ENABLE_SSL_V3=(false|0,true|1)
45 // LIBPROCESS_SSL_ENABLE_TLS_V1_0=(false|0,true|1)
46 // LIBPROCESS_SSL_ENABLE_TLS_V1_1=(false|0,true|1)
47 // LIBPROCESS_SSL_ENABLE_TLS_V1_2=(false|0,true|1)
48 //
